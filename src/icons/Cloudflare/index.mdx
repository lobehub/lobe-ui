title: Cloudflare
description: Re-exports the Cloudflare icon and avatar variants from @lobehub/icons.
category: Auth Icons
---

## Icons

```tsx
import { Cloudflare } from '@lobehub/ui/icons';

import { Flexbox } from '@/Flex';

export default () => <Cloudflare size={64} />;
```

## Avatars

```tsx
import { Cloudflare } from '@lobehub/ui/icons';

import { Flexbox } from '@/Flex';

export default () => (
  <Flexbox gap={16} horizontal>
    <Cloudflare.Avatar size={64} />
    <Cloudflare.Avatar size={64} shape={'square'} />
  </Flexbox>
);
```
